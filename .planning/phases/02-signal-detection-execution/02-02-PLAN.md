   ```mql5
   Setup1Signal DetectSetup1Signal() {
       Setup1Signal result = {false, false, 0, 0, 0, 0, 0, 0};
       
       // REQ-011: Balanced market check (caller checks via IsBalancedMarket())
       
       // REQ-012: Gap detection — price opened outside previous session VA
       double prevSessionVAH = currentProfile.vahPrice;  // Previous session (stored in profile)
       double prevSessionVAL = currentProfile.valPrice;
       
       double openPrice = iOpen(Symbol(), PERIOD_CURRENT, 1);  // Previous closed bar
       double closePrice = iClose(Symbol(), PERIOD_CURRENT, 1);
       double lowPrice = iLow(Symbol(), PERIOD_CURRENT, 1);
       
       bool gappedAboveVA = (openPrice > prevSessionVAH);
       bool gappedBelowVA = (openPrice < prevSessionVAL);
       
       if (!gappedAboveVA && !gappedBelowVA)
           return result;  // No gap; skip
       
       // REQ-013: Reclaim detection — price reclaimed into VA on current bar
       bool reclaimingUp = (gappedBelowVA && closePrice >= prevSessionVAL);
       bool reclaimingDown = (gappedAboveVA && closePrice <= prevSessionVAH);
       
       if (!reclaimingUp && !reclaimingDown)
           return result;  // No reclaim; skip
       
       // REQ-014: Confirmation candle — close FULLY inside VA (not wick touch)
       bool closeInsideVA = (closePrice >= prevSessionVAL && closePrice <= prevSessionVAH);
       
       if (!closeInsideVA)
           return result;  // Wick touch or rejection; skip
       
       // Signal triggered!
       result.isTriggered = true;
       result.isLong = reclaimingUp;  // LONG if reclaiming from below
       result.confirmationClose = closePrice;
       result.sweepLow = lowPrice;
       
       // REQ-015/REQ-016: Entry price = confirmation close (market order at bar close)
       result.entryPrice = closePrice;
       
       // REQ-026: SL placement = 5-15 pips below sweep low (use 10 pips as default)
       result.stopLoss = lowPrice - (10 * Point);
       
       // D-03: TP = opposite VA extreme (single unified TP, not 65%/35% split)
       result.takeProfit = result.isLong ? prevSessionVAH : prevSessionVAL;
       
       // REQ-028: Calculate Risk/Reward ratio
       double slDistance = MathAbs(result.entryPrice - result.stopLoss) / Point;
       double tpDistance = MathAbs(result.takeProfit - result.entryPrice) / Point;
       result.riskRewardRatio = (slDistance > 0) ? tpDistance / slDistance : 0;
       
       return result;
   }
   ```

   ```mql5
   enum CandlePatternType { PATTERN_NONE = 0, PATTERN_HAMMER = 1, PATTERN_SHOOTING_STAR = 2, PATTERN_DOJI = 3 };
   
   CandlePatternType DetectCandlePattern() {
       // REQ-019: Pattern recognition (Hammer, Shooting Star, Doji) on bar [1]
       double open = iOpen(Symbol(), PERIOD_CURRENT, 1);
       double high = iHigh(Symbol(), PERIOD_CURRENT, 1);
       double low = iLow(Symbol(), PERIOD_CURRENT, 1);
       double close = iClose(Symbol(), PERIOD_CURRENT, 1);
       
       double bodySize = MathAbs(close - open);
       double lowerWick = open < close ? open - low : close - low;
       double upperWick = close > open ? high - close : high - open;
       
       // HAMMER: Lower wick > 2x body, upper wick < 0.1x body, close near high
       if (lowerWick > 2 * bodySize && upperWick < 0.1 * bodySize && close > (open + bodySize * 0.5))
           return PATTERN_HAMMER;
       
       // SHOOTING STAR: Upper wick > 2x body, lower wick < 0.1x body, close near low
       if (upperWick > 2 * bodySize && lowerWick < 0.1 * bodySize && close < (open - bodySize * 0.5))
           return PATTERN_SHOOTING_STAR;
       
       // DOJI: Open ≈ close (within 1 pip), wicks extending both sides
       if (bodySize <= 1 * Point && lowerWick > 0 && upperWick > 0)
           return PATTERN_DOJI;
       
       return PATTERN_NONE;
   }
   ```

2. Add Setup 2 detection function:
   ```mql5
   Setup2Signal DetectSetup2Signal() {
       Setup2Signal result = {false, false, 0, 0, 0, 0, 0, 0};
       
       // REQ-017: LVN sweep detection — recent low below lowest LVN
       double lowestLVN = 999999;
       if (currentProfile.lvnCount > 0) {
           for (int i = 0; i < currentProfile.lvnCount; i++) {
               if (currentProfile.lvnArray[i].price < lowestLVN)
                   lowestLVN = currentProfile.lvnArray[i].price;
           }
       }
       
       double currentLow = iLow(Symbol(), PERIOD_CURRENT, 1);
       if (currentLow > lowestLVN)
           return result;  // No LVN sweep; skip
       
       // REQ-018: HVN edge identification — find nearest HVN above current price
       double hvnEdge = 999999;
       if (currentProfile.hvnCount > 0) {
           for (int i = 0; i < currentProfile.hvnCount; i++) {
               if (currentProfile.hvnArray[i].price > currentLow && 
                   currentProfile.hvnArray[i].price < hvnEdge)
               {
                   hvnEdge = currentProfile.hvnArray[i].price;
               }
           }
       }
       
       if (hvnEdge == 999999)
           return result;  // No HVN edge found; skip
       
       // REQ-019: Trigger pattern recognition (Hammer/Shooting Star/Doji)
       CandlePatternType pattern = DetectCandlePattern();
       if (pattern == PATTERN_NONE)
           return result;  // No valid pattern; skip
       
       // REQ-020: Volume spike confirmation (≥ 1.3x previous bar) — locked threshold per D-04
       long currentVolume = iVolume(Symbol(), PERIOD_CURRENT, 1);
       long previousVolume = iVolume(Symbol(), PERIOD_CURRENT, 2);
       
       if (previousVolume <= 0 || currentVolume < previousVolume * 1.3)
           return result;  // Insufficient volume; skip (1.3x threshold locked per D-04)
       
       // REQ-021: Closed candle requirement (already using bar [1], not [0])
       
       // Signal triggered!
       result.isTriggered = true;
       result.isLong = (pattern == PATTERN_HAMMER);  // LONG on Hammer, SHORT on Shooting Star
       result.hvnEdgePrice = hvnEdge;
       result.sweepLow = currentLow;  // Used for SL (below LVN)
       
       // REQ-022/REQ-023: Entry price = HVN edge price (market order at bar close)
       result.entryPrice = hvnEdge;
       
       // REQ-026: SL placement = 5-15 pips below LVN sweep (use 10 pips)
       result.stopLoss = result.sweepLow - (10 * Point);
       
       // D-06: TP = opposite profile boundary (VAH for LONG, VAL for SHORT)
       result.takeProfit = result.isLong ? currentProfile.vahPrice : currentProfile.valPrice;
       
       // REQ-028: Calculate Risk/Reward ratio
       double slDistance = MathAbs(result.entryPrice - result.stopLoss) / Point;
       double tpDistance = MathAbs(result.takeProfit - result.entryPrice) / Point;
       result.riskRewardRatio = (slDistance > 0) ? tpDistance / slDistance : 0;
       
       return result;
   }
   ```
