  ```mql5
  // In RiskLimits.mqh

  struct DailyLimitState
  {
    double closedPnL;
    double openPnL;
    double totalPnL;
    bool hardStopHit;
    bool profitCapReached;
  };

  DailyLimitState dailyLimits = {0, 0, 0, false, false};

  DailyLimitState CalculateDailyPnL()
  {
    // Rescan OrdersHistoryTotal every tick (persistent across EA restart)
    DailyLimitState result = {0, 0, 0, false, false};
    
    datetime sessionStart = GetSessionBoundary();  // From Utils.mqh
    
    // Scan closed trades closed since session start (today)
    for (int i = OrdersHistoryTotal() - 1; i >= 0; i--)
    {
      if (!OrderSelect(i, SELECT_BY_POS, MODE_HISTORY))
        continue;
      
      // Filter for this EA's trades (magic number)
      if (OrderMagicNumber() != EA_MAGIC_NUMBER && 
          !(OrderMagicNumber() >= EA_MAGIC_NUMBER && OrderMagicNumber() < EA_MAGIC_NUMBER + 10))
        continue;
      
      // Include only trades closed in current session
      if (OrderCloseTime() >= sessionStart)
      {
        result.closedPnL += OrderProfit() + OrderCommission() + OrderSwap();
      }
    }
    
    // Scan open positions for current P&L
    for (int i = 0; i < positionCount; i++)  // positionCount from TradeExecution.mqh
    {
      long ticket = positions[i].ticket;
      
      // Get position P&L
      double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
      double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
      
      double pnl = 0;
      if (positions[i].isLong)
        pnl = (bid - positions[i].entryPrice) * positions[i].remainingLots * SymbolInfoDouble(Symbol(), SYMBOL_TRADE_TICK_VALUE);
      else
        pnl = (positions[i].entryPrice - ask) * positions[i].remainingLots * SymbolInfoDouble(Symbol(), SYMBOL_TRADE_TICK_VALUE);
      
      result.openPnL += pnl;
    }
    
    result.totalPnL = result.closedPnL + result.openPnL;
    
    return result;
  }

  bool EnforceDailyLimits()
  {
    // Recalculate daily P&L
    DailyLimitState limits = CalculateDailyPnL();
    dailyLimits = limits;
    
    double accountBalance = AccountBalance();
    double hardStopThreshold = accountBalance * (DAILY_LOSS_LIMIT / 100.0);     // -2%
    double profitCapThreshold = accountBalance * (DAILY_PROFIT_CAP / 100.0);    // +5%
    
    // D-09: Hard stop loss (-2%)
    if (limits.totalPnL < -hardStopThreshold)
    {
      if (!limits.hardStopHit)  // Log once
      {
        LogAlert("HARD_STOP_HIT", 
                StringFormat("Daily loss=%.2f, Limit=-%.2f", limits.totalPnL, hardStopThreshold));
      }
      
      // Force-close ALL positions
      for (int i = positionCount - 1; i >= 0; i--)
      {
        trade.PositionClose(positions[i].ticket);
        ClosePosition(positions[i].ticket, 0, "HARD_STOP", positions[i].remainingLots);
      }
      
      return false;  // Block new entries
    }
    
    // D-10: Daily profit cap (+5%)
    if (limits.totalPnL > profitCapThreshold)
    {
      if (!limits.profitCapReached)  // Log once
      {
        LogAlert("PROFIT_CAP_REACHED",
                StringFormat("Daily profit=%.2f, Cap=+%.2f", limits.totalPnL, profitCapThreshold));
      }
      
      // Close 50-70% of positions (use 60% midpoint)
      int closeCount = (int)MathCeil(positionCount * 0.6);
      for (int i = 0; i < closeCount && i < positionCount; i++)
      {
        trade.PositionClose(positions[i].ticket);
        ClosePosition(positions[i].ticket, 0, "PROFIT_CAP_CLOSE", positions[i].remainingLots);
      }
      
      // Move SL to profit on remaining positions (breakeven + 5 pips)
      for (int i = closeCount; i < positionCount; i++)
      {
        double newSL = positions[i].entryPrice;  // Breakeven
        if (positions[i].isLong)
          newSL += 5 * Point;  // +5 pips profit
        else
          newSL -= 5 * Point;
        
        positions[i].stopLoss = newSL;
        
        // Update order via CTrade
        MqlTradeRequest request = {0};
        request.action = TRADE_ACTION_SLTP;
        request.symbol = Symbol();
        request.position = positions[i].ticket;
        request.sl = newSL;
        request.tp = positions[i].takeProfit;
        trade.Send(request);
      }
      
      return false;  // Block new entries
    }
    
    return true;  // Trading allowed
  }
  ```

  ```mql5
  bool CheckFridayHardClose()
  {
    // Get current time in broker server time
    datetime currentTime = TimeCurrent();
    MqlDateTime timeStruct;
    TimeToStruct(currentTime, timeStruct);
    
    // Check if Friday (day of week: 0=Sunday, 5=Friday, 6=Saturday)
    bool isFriday = (timeStruct.day_of_week == 5);
    
    // Check if time >= 21:45
    int currentTimeMinutes = timeStruct.hour * 60 + timeStruct.min;
    int closeTimeMinutes = 21 * 60 + 45;  // 21:45 = 1305 minutes
    bool isCloseTime = (currentTimeMinutes >= closeTimeMinutes);
    
    if (isFriday && isCloseTime)
    {
      // Force-close ALL open positions
      if (positionCount > 0)
      {
        LogAlert("FRIDAY_HARD_CLOSE", 
                StringFormat("Time=%02d:%02d, Closing %d positions",
                            timeStruct.hour, timeStruct.min, positionCount));
      }
      
      // Close all positions
      for (int i = positionCount - 1; i >= 0; i--)
      {
        trade.PositionClose(positions[i].ticket);
        ClosePosition(positions[i].ticket, 0, "FRIDAY_CLOSE", positions[i].remainingLots);
      }
      
      return true;  // Hard close executed

    return false;
  ```

  ```mql5
  // In ReversalExit.mqh

  struct ReversalSignal
  {
    bool isTriggered;
    bool isReversal;     // true = confirmed reversal
    bool isLong;         // direction of reversal
    double reversalPrice; // Price level of reversal candle
  };

  ReversalSignal DetectReversalCandle(bool currentLong)
  {
    ReversalSignal result = {false, false, false, 0};
    
    // Detect 5M reversal candle at TP level
    // For LONG position: lower high (rejection of VAH)
    // For SHORT position: higher low (rejection of VAL)
    
    double currentHigh = iHigh(Symbol(), PERIOD_CURRENT, 1);  // Previous 5M bar
    double currentLow = iLow(Symbol(), PERIOD_CURRENT, 1);
    double previousHigh = iHigh(Symbol(), PERIOD_CURRENT, 2);
    double previousLow = iLow(Symbol(), PERIOD_CURRENT, 2);
    
    if (currentLong) {
      // LONG position reversal: lower high (rejection of resistance/VAH)
      if (currentHigh < previousHigh) {
        result.isTriggered = true;
        result.isLong = false;  // Reversal is SHORT direction
        result.reversalPrice = currentHigh;
        return result;
      }
    } else {
      // SHORT position reversal: higher low (rejection of support/VAL)
      if (currentLow > previousLow) {
        result.isTriggered = true;
        result.isLong = true;   // Reversal is LONG direction
        result.reversalPrice = currentLow;
        return result;
      }
    }
    
    return result;
  }

  bool ConfirmReversal1M(bool reversalIsLong)
  {
    // Confirm reversal on 1M timeframe
    // For LONG reversal: break of recent 1M high
    // For SHORT reversal: break of recent 1M low
    
    double high1M = iHighest(Symbol(), PERIOD_M1, MODE_HIGH, 5, 0);  // Highest in last 5 1M bars
    double low1M = iLowest(Symbol(), PERIOD_M1, MODE_LOW, 5, 0);
    
    double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
    double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
    
    if (reversalIsLong) {
      // LONG reversal: price must break above 1M recent high + some buffer
      return (ask > high1M + 10 * Point);
    } else {
      // SHORT reversal: price must break below 1M recent low - some buffer
      return (bid < low1M - 10 * Point);
    }
  }

  bool ExecutePositionFlip(long oldTicket, bool newLongEntry, double newEntryPrice,
                           double newStopLoss, double newTakeProfit)
  {
    // Close current position + Enter new position in opposite direction
    
    // Step 1: Close current position
    if (!trade.PositionClose(oldTicket)) {
      LogError(StringFormat("Failed to close position %d for flip", oldTicket));
      return false;
    }
    
    int oldIdx = FindPositionByTicket(oldTicket);
    if (oldIdx >= 0) {
      ClosePosition(oldTicket, newEntryPrice, "REVERSAL_EXIT", positions[oldIdx].remainingLots);
    }
    
    // Step 2: Enter new position (opposite direction)
    double lotSize = CalculateLotSize(newEntryPrice, newStopLoss);
    
    OrderResult result = PlaceMarketOrder(
      newLongEntry ? ORDER_TYPE_BUY : ORDER_TYPE_SELL,
      lotSize,
      newEntryPrice,
      newStopLoss,
      newTakeProfit);
    
    if (result.success) {
      double rr = CalculateRiskRewardRatio(result.fillPrice, newStopLoss, newTakeProfit);
      AddPosition(result.ticket, Symbol(), newLongEntry, result.fillPrice,
                  newStopLoss, newTakeProfit, lotSize, "REVERSAL", rr);
      
      LogAlert("POSITION_FLIP", 
              StringFormat("Old=%d closed at REVERSAL. New=%d entered %s at %.5f",
                          oldTicket, result.ticket, newLongEntry ? "LONG" : "SHORT", result.fillPrice));
      
      return true;
    }
    
    return false;
  }
  ```
