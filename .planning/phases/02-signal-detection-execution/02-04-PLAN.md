   bool EnforceDailyHardStop() {
       // REQ-035: Persistent P&L calculation (rescan OrdersHistoryTotal every tick)
       double closedPnL = 0;
       
       // Scan closed trades from TODAY
       for (int i = OrdersHistoryTotal() - 1; i >= 0; i--) {
           if (!HistorySelectByPosition(i))
               continue;
           
           // Check if closed today (within last 24 hours from Tokyo open)
           if (TimeCurrent() - HistoryOrderGetInteger(ORDER_TIME_DONE) < 86400) {
               if (HistoryOrderGetInteger(ORDER_MAGIC) >= 99001 && 
                   HistoryOrderGetInteger(ORDER_MAGIC) < 99010) {  // This EA's magic range
                   closedPnL += HistoryOrderGetDouble(ORDER_PROFIT);
               }
           }
       }
       
       // Calculate open position P&L
       double openPnL = 0;
       for (int i = 0; i < positionCount; i++) {
           double positionProfit = 0;
           if (PositionSelect(positions[i].symbol)) {
               positionProfit = PositionGetDouble(POSITION_PROFIT);
           }
           openPnL += positionProfit;
       }
       
       double totalPnL = closedPnL + openPnL;
       double accountBalance = AccountBalance();
       double hardStopThreshold = accountBalance * 0.02;  // -2% per D-09 (locked)
       
       // REQ-032: Hard stop enforcement — non-override, immediate action
       if (totalPnL < -hardStopThreshold) {
           LogAlert("HARD_STOP_HIT",
                   StringFormat("Daily PnL=%.2f, Threshold=-%.2f", totalPnL, hardStopThreshold));
           
           // Force-close ALL open positions immediately
           for (int i = positionCount - 1; i >= 0; i--) {
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("HARD_STOP_CLOSE",
                           StringFormat("Ticket=%lld, Reason=Daily Hard Stop", positions[i].ticket));
                   RemovePosition(i);
               }
           }
           
           return false;  // Trading blocked for remainder of session
       }
       
       return true;  // Trading allowed
   }
   ```

   bool EnforceDailyProfitCap() {
       // Same P&L calculation as hard stop
       double closedPnL = 0;
       
       for (int i = OrdersHistoryTotal() - 1; i >= 0; i--) {
           if (!HistorySelectByPosition(i))
               continue;
           
           if (TimeCurrent() - HistoryOrderGetInteger(ORDER_TIME_DONE) < 86400) {
               if (HistoryOrderGetInteger(ORDER_MAGIC) >= 99001 && 
                   HistoryOrderGetInteger(ORDER_MAGIC) < 99010) {
                   closedPnL += HistoryOrderGetDouble(ORDER_PROFIT);
               }
           }
       }
       
       double openPnL = 0;
       for (int i = 0; i < positionCount; i++) {
           double positionProfit = 0;
           if (PositionSelect(positions[i].symbol)) {
               positionProfit = PositionGetDouble(POSITION_PROFIT);
           }
           openPnL += positionProfit;
       }
       
       double totalPnL = closedPnL + openPnL;
       double accountBalance = AccountBalance();
       double profitCapThreshold = accountBalance * 0.05;  // +5% per D-10 (locked)
       
       // REQ-033: Profit cap enforcement (more flexible than hard stop)
       if (totalPnL > profitCapThreshold) {
           LogAlert("PROFIT_CAP_REACHED",
                   StringFormat("Daily PnL=%.2f, Cap=+%.2f", totalPnL, profitCapThreshold));
           
           // Close 50-70% of positions (use 60% as midpoint per D-10)
           int closeCount = (int)MathCeil(positionCount * 0.6);
           
           for (int i = 0; i < closeCount && i < positionCount; i++) {
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("PROFIT_CAP_CLOSE",
                           StringFormat("Ticket=%lld, Closed=60%% of position", positions[i].ticket));
                   RemovePosition(i);
               }
           }
           
           // Move SL of remaining positions to profit (breakeven + 5 pips per D-10)
           for (int i = 0; i < positionCount; i++) {
               double newSL = positions[i].entryPrice;
               
               if (positions[i].isLong)
                   newSL += 5 * Point;  // +5 pips for LONG
               else
                   newSL -= 5 * Point;  // -5 pips for SHORT
               
               // Update SL via CTrade SLTP action
               MqlTradeRequest request = {0};
               request.action = TRADE_ACTION_SLTP;
               request.symbol = positions[i].symbol;
               request.position = positions[i].ticket;
               request.sl = newSL;
               request.tp = positions[i].takeProfit;
               
               MqlTradeResult result = {0};
               if (trade.Send(request, result)) {
                   positions[i].stopLoss = newSL;  // Update tracking
                   LogAlert("SL_MOVED_TO_PROFIT",
                           StringFormat("Ticket=%lld, NewSL=%.2f", positions[i].ticket, newSL));
               }
           }
           
           return false;  // Block new entries
       }
       
       return true;  // New entries allowed
   }
   ```

   bool CheckFridayClose() {
       // REQ-034: Check if Friday >= 21:45 broker server time (locked per D-11)
       datetime currentTime = TimeCurrent();
       MqlDateTime timeStruct;
       TimeToStruct(currentTime, timeStruct);
       
       // Check if Friday (day_of_week: 5 = Friday)
       bool isFriday = (timeStruct.day_of_week == 5);
       
       // Check if time >= 21:45 (21*60+45 = 1305 minutes)
       int currentTimeMinutes = timeStruct.hour * 60 + timeStruct.min;
       int closeTimeMinutes = 21 * 60 + 45;  // 21:45
       bool isCloseTime = (currentTimeMinutes >= closeTimeMinutes);
       
       if (isFriday && isCloseTime) {
           if (positionCount > 0) {
               LogAlert("FRIDAY_HARD_CLOSE",
                       StringFormat("Time=%d:%02d, Closing %d positions",
                                   timeStruct.hour, timeStruct.min, positionCount));

           // Force-close ALL open positions
           for (int i = positionCount - 1; i >= 0; i--) {
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("FRIDAY_CLOSE_EXECUTED",
                           StringFormat("Ticket=%lld", positions[i].ticket));
                   RemovePosition(i);
               }
           }
           
           return true;  // Friday close executed
