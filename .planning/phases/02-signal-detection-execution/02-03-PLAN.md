   ```mql5
   struct OrderResult {
       bool success;
       long ticket;
       double fillPrice;
       double slippage;  // Actual slippage in pips
   };
   
   OrderResult PlaceMarketOrder(bool isLong, double lots, double entryPrice, 
                                 double stopLoss, double takeProfit) {
       OrderResult result = {false, 0, 0, 0};
       
       // REQ-040: Verify broker connectivity before order placement
       if (!IsConnected()) {
           LogAlert("ORDER_REJECTED", "Broker not connected");
           return result;
       }
       
       // REQ-041: Retry logic — up to 3 attempts with exponential backoff
       for (int attempt = 0; attempt < 3; attempt++) {
           // Prepare trade request
           MqlTradeRequest request = {0};
           request.action = TRADE_ACTION_DEAL;
           request.symbol = Symbol();
           request.volume = lots;
           request.type = isLong ? ORDER_TYPE_BUY : ORDER_TYPE_SELL;
           request.price = entryPrice;
           request.sl = stopLoss;
           request.tp = takeProfit;
           request.deviation = 500;  // 50 pips (5 decimals = 500 points)
           request.magic = 99001;    // EA magic number (unique ID for this EA)
           request.comment = "Setup" + (IsBalancedMarket() ? "1" : "2");
           
           // Execute via CTrade
           MqlTradeResult tradeResult = {0};
           if (!trade.Send(request, tradeResult)) {
               uint retcode = tradeResult.retcode;
               LogAlert("ORDER_SEND_FAILED", 
                       StringFormat("Attempt %d, Error Code %d", attempt + 1, retcode));
               
               // Retry on transient errors
               if (attempt < 2) {
                   Sleep(100);  // Wait before retry
                   continue;
               } else {
                   result.success = false;
                   return result;
               }
           }
           
           uint retcode = tradeResult.retcode;
           
           // Check for successful execution
           if (retcode == TRADE_RETCODE_DONE || retcode == TRADE_RETCODE_PLACED) {
               result.ticket = tradeResult.order;
               result.fillPrice = tradeResult.price;
               
               // D-07: Validate slippage (50 pip tolerance per CONTEXT.md)
               double slippagePips = MathAbs(result.fillPrice - entryPrice) / Point;
               
               if (slippagePips <= 50.0) {
                   // Slippage acceptable
                   result.success = true;
                   result.slippage = slippagePips;
                   
                   LogAlert("ORDER_PLACED", 
                           StringFormat("Ticket=%lld, Price=%.2f, Slippage=%.1f pips",
                                       result.ticket, result.fillPrice, slippagePips));
                   
                   return result;
               } else {
                   // REQ-039: Slippage exceeds 50 pips; reject and close immediately
                   LogAlert("ORDER_REJECTED", 
                           StringFormat("Slippage %.1f pips exceeds 50 pip limit", slippagePips));
                   
                   // Close the erroneous position immediately
                   trade.PositionClose(result.ticket);
                   
                   result.success = false;
                   return result;
               }
           } else {
               LogAlert("ORDER_FAILED", 
                       StringFormat("Retcode %d, Attempt %d", retcode, attempt + 1));
               
               if (attempt < 2) {
                   Sleep(100);
                   continue;
               } else {
                   result.success = false;
                   return result;
               }
           }
       }
       
       return result;
   }
   ```

   ```mql5
   // REQ-031: Check if max 1 position per asset already open
   bool IsMaxPositionExceeded(string symbol) {
       for (int i = 0; i < positionCount; i++) {
           if (positions[i].symbol == symbol)
               return true;  // Position already open on this symbol
       }
       return false;  // No position open
   }
   
   // Add new position to tracking array
   bool AddPosition(long ticket, bool isLong, double entryPrice, double stopLoss, 
                    double takeProfit, double originalLots, string setupType, double riskRewardRatio) {
       if (positionCount >= MAX_POSITIONS)
           return false;  // Array full
       
       positions[positionCount].ticket = ticket;
       positions[positionCount].symbol = Symbol();
       positions[positionCount].isLong = isLong;
       positions[positionCount].entryPrice = entryPrice;
       positions[positionCount].stopLoss = stopLoss;
       positions[positionCount].takeProfit = takeProfit;
       positions[positionCount].originalLots = originalLots;
       positions[positionCount].remainingLots = originalLots;  // Initially full
       positions[positionCount].entryTime = TimeCurrent();
       positions[positionCount].setupType = setupType;
       positions[positionCount].riskRewardRatio = riskRewardRatio;
       
       positionCount++;
       return true;
   }
   
   // REQ-027: Update position state when partial close occurs
   bool UpdatePositionState(long ticket, double partialCloseLots) {
       for (int i = 0; i < positionCount; i++) {
           if (positions[i].ticket == ticket) {
               positions[i].remainingLots -= partialCloseLots;
               
               if (positions[i].remainingLots <= 0) {
                   RemovePosition(i);
                   return true;
               }
               return true;
           }
       }
       return false;  // Position not found
   }
   
   // Remove position from tracking array
   void RemovePosition(int index) {
       for (int i = index; i < positionCount - 1; i++) {
           positions[i] = positions[i + 1];
       }
       positionCount--;
   }
   ```

   ```mql5
   void MonitorPositionExits() {
       // Check all open positions every tick
       for (int i = positionCount - 1; i >= 0; i--) {
           // Get current market price
           double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
           double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
           
           // Check TP hit (full position close at TP)
           if (positions[i].isLong && bid >= positions[i].takeProfit) {
               // TP hit on LONG; close entire remaining position
               double closeLots = positions[i].remainingLots;
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("POSITION_CLOSED", 
                           StringFormat("Ticket=%lld, Reason=TP, Lots=%.2f",
                                       positions[i].ticket, closeLots));
                   
                   RemovePosition(i);
               }
               continue;
           }
           
           if (!positions[i].isLong && ask <= positions[i].takeProfit) {
               // TP hit on SHORT; close entire remaining position
               double closeLots = positions[i].remainingLots;
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("POSITION_CLOSED",
                           StringFormat("Ticket=%lld, Reason=TP, Lots=%.2f",
                                       positions[i].ticket, closeLots));
                   
                   RemovePosition(i);
               }
               continue;
           }
           
           // Check SL hit (position closed at SL)
           if (positions[i].isLong && bid <= positions[i].stopLoss) {
               // SL hit on LONG; close position
               double closeLots = positions[i].remainingLots;
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("POSITION_CLOSED",
                           StringFormat("Ticket=%lld, Reason=SL, Lots=%.2f",
                                       positions[i].ticket, closeLots));
                   
                   RemovePosition(i);
               }
               continue;
           }
           
           if (!positions[i].isLong && ask >= positions[i].stopLoss) {
               // SL hit on SHORT; close position
               double closeLots = positions[i].remainingLots;
               bool closeSuccess = trade.PositionClose(positions[i].ticket);
               
               if (closeSuccess) {
                   LogAlert("POSITION_CLOSED",
                           StringFormat("Ticket=%lld, Reason=SL, Lots=%.2f",
                                       positions[i].ticket, closeLots));
                   
                   RemovePosition(i);
               }
               continue;
           }
       }
   }
   ```
