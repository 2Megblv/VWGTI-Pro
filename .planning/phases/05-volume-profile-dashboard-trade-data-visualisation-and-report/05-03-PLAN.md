```mql5
void RenderDashboard()
{
    long chartId = ChartID();
    int xPos = DashboardXOffset;      // Left edge offset (extern variable)
    int yPos = DashboardYOffset;      // Top edge offset
    const int lineHeight = 20;        // Pixels between lines
    const color titleColor = clrWhite;
    const color statsColor = clrLimeGreen;
    const color warningColor = clrOrange;
    const color errorColor = clrRed;
    
    // === SECTION 1: HEADER ===
    UpdateDashboardLabel(chartId, "Dashboard_Title", xPos, yPos, 
                        "VOLUME PROFILE DASHBOARD", titleColor, 12);
    yPos += lineHeight * 1.5;
    
    // === SECTION 2: EQUITY & DAILY P&L ===
    // Current Equity
    double currentEquity = g_metrics.currentEquity;
    UpdateDashboardLabel(chartId, "Dashboard_Equity", xPos, yPos,
                        StringFormat("Equity: $%.2f", currentEquity), 
                        statsColor, 10);
    yPos += lineHeight;
    
    // Daily P&L
    double dailyPnL = 0;
    if (g_metrics.dailyCount > 0)
    {
        dailyPnL = g_metrics.dailyPnLValues[g_metrics.dailyCount - 1];
    }
    double dailyPnLPercent = (currentEquity > 0) 
        ? (dailyPnL / currentEquity) * 100 
        : 0;
    color pnlColor = (dailyPnL >= 0) ? clrLimeGreen : clrRed;
    
    UpdateDashboardLabel(chartId, "Dashboard_DailyPnL", xPos, yPos,
                        StringFormat("Daily P&L: $%.2f (%.2f%%)", dailyPnL, dailyPnLPercent),
                        pnlColor, 10);
    yPos += lineHeight * 1.5;
    
    // === SECTION 3: SUMMARY STATS HEADER ===
    UpdateDashboardLabel(chartId, "Dashboard_StatsHeader", xPos, yPos,
                        "SUMMARY STATS", titleColor, 11);
    yPos += lineHeight;
    
    // === SECTION 4: SUMMARY STATS VALUES ===
    
    // Total Trades
    UpdateDashboardLabel(chartId, "Dashboard_TotalTrades", xPos, yPos,
                        StringFormat("Total Trades: %d", g_metrics.totalTrades),
                        statsColor, 10);
    yPos += lineHeight;
    
    // Win Rate (Phase 3 gate: ≥50%)
    color wrColor = (g_metrics.winRate >= MIN_WIN_RATE) ? clrLimeGreen : clrRed;
    string wrCheck = (g_metrics.winRate >= MIN_WIN_RATE) ? " ✓" : " ✗";
    UpdateDashboardLabel(chartId, "Dashboard_WinRate", xPos, yPos,
                        StringFormat("Win Rate: %.1f%%%s", g_metrics.winRate * 100, wrCheck),
                        wrColor, 10);
    yPos += lineHeight;
    
    // Profit Factor (Phase 3 gate: ≥1.5)
    color pfColor = (g_metrics.profitFactor >= MIN_PROFIT_FACTOR) ? clrLimeGreen : clrRed;
    string pfCheck = (g_metrics.profitFactor >= MIN_PROFIT_FACTOR) ? " ✓" : " ✗";
    UpdateDashboardLabel(chartId, "Dashboard_ProfitFactor", xPos, yPos,
                        StringFormat("Profit Factor: %.2f%s", g_metrics.profitFactor, pfCheck),
                        pfColor, 10);
    yPos += lineHeight;
    
    // Max Daily Drawdown (Phase 3 gate: ≤2%)
    color ddColor = (g_metrics.maxDailyDrawdown <= MAX_DAILY_DRAWDOWN) ? clrLimeGreen : clrRed;
    string ddCheck = (g_metrics.maxDailyDrawdown <= MAX_DAILY_DRAWDOWN) ? " ✓" : " ✗";
    UpdateDashboardLabel(chartId, "Dashboard_MaxDD", xPos, yPos,
                        StringFormat("Max DD: %.2f%%%s", g_metrics.maxDailyDrawdown * 100, ddCheck),
                        ddColor, 10);
    yPos += lineHeight * 1.5;
    
    // === SECTION 5: PER-SYMBOL BREAKDOWN ===
    UpdateDashboardLabel(chartId, "Dashboard_SymbolHeader", xPos, yPos,
                        "XAUUSD", titleColor, 11);
    
    // XAUUSD positioned at xPos
    int xauColumn = xPos;
    int eurColumn = xPos + 180;  // 180 pixels to the right for EURUSD column
    int symYPos = yPos;
    
    // XAUUSD data
    UpdateDashboardLabel(chartId, "Dashboard_XAUUSD_Trades", xauColumn, symYPos,
                        StringFormat("Trades: %d", g_metrics.symbolXAUUSD_trades),
                        statsColor, 10);
    symYPos += lineHeight;
    
    UpdateDashboardLabel(chartId, "Dashboard_XAUUSD_WinRate", xauColumn, symYPos,
                        StringFormat("Win Rate: %.1f%%", g_metrics.symbolXAUUSD_winRate * 100),
                        statsColor, 10);
    symYPos += lineHeight;
    
    UpdateDashboardLabel(chartId, "Dashboard_XAUUSD_PF", xauColumn, symYPos,
                        StringFormat("Profit Factor: %.2f", g_metrics.symbolXAUUSD_profitFactor),
                        statsColor, 10);
    symYPos += lineHeight;
    
    UpdateDashboardLabel(chartId, "Dashboard_XAUUSD_PnL", xauColumn, symYPos,
                        StringFormat("Total P&L: $%.2f", g_metrics.symbolXAUUSD_totalPnL),
                        statsColor, 10);
    
    // EURUSD data (positioned to the right)
    symYPos = yPos;
    
    UpdateDashboardLabel(chartId, "Dashboard_EURUSD_Header", eurColumn, yPos,

}
```

```mql5
//+------------------------------------------------------------------+
//| Update Dashboard Label (create or update ChartObject label)      |
//| Helper function to handle object creation and text updates       |
//|                                                                  |
//| Pitfall Avoidance (from RESEARCH.md Pitfall 5):                |
//|   - Check if object exists before creating                       |
//|   - Use ObjectSetString() to update existing objects             |
//|   - Call ChartRedraw() to force display update                   |
//+------------------------------------------------------------------+

void UpdateDashboardLabel(long chartId, string objectName, int xDistance, int yDistance,
                          string text, color textColor = clrWhite, int fontSize = 10)
{
    // Check if object already exists
    if (ObjectFind(chartId, objectName) >= 0)
    {
        // Object exists; just update the text
        ObjectSetString(chartId, objectName, OBJPROP_TEXT, text);
        ObjectSetInteger(chartId, objectName, OBJPROP_COLOR, textColor);
        // No need to reposition; position only changes when moved
    }
    else
    {
        // Object doesn't exist; create it
        if (!ObjectCreate(chartId, objectName, OBJ_LABEL, 0, 0, 0))
        {
            Print(StringFormat("Failed to create label: %s", objectName));
            return;
        }
        
        // Set position
        ObjectSetInteger(chartId, objectName, OBJPROP_XDISTANCE, xDistance);
        ObjectSetInteger(chartId, objectName, OBJPROP_YDISTANCE, yDistance);
        
        // Set text and style
        ObjectSetString(chartId, objectName, OBJPROP_TEXT, text);
        ObjectSetInteger(chartId, objectName, OBJPROP_COLOR, textColor);
        ObjectSetInteger(chartId, objectName, OBJPROP_FONTSIZE, fontSize);
        ObjectSetString(chartId, objectName, OBJPROP_FONT, "Arial");
        
        // Additional styling (optional)
        ObjectSetInteger(chartId, objectName, OBJPROP_ANCHOR, ANCHOR_LEFT_TOP);
        ObjectSetInteger(chartId, objectName, OBJPROP_BACK, false);  // Don't block mouse clicks
    }
}
```

**Key Pattern (from RESEARCH.md Pattern 3):**

- **First check:** `ObjectFind(chartId, objectName) >= 0` — Does object exist?
- **If exists:** Update text and color only (faster, less flicker)
- **If doesn't exist:** Create object, set position, set style
- **Always use ObjectSetString():** For text updates (much faster than delete+recreate)
- **ChartRedraw():** Called at end of RenderDashboard() to force redraw
