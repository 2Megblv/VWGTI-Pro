```mql5
//| Volume Profile Dashboard — Real-Time Performance Panel           |
//| Phase 5: Trade Data Visualisation and Reporting                  |
//|                                                                  |
//| Purpose:                                                         |
//|   Dedicated indicator running on separate chart window.          |
//|   Updates equity curve, P&L metrics, and summary stats on every  |
//|   bar close. Displays via ChartObjects.                          |
//|                                                                  |
//| Integration:                                                     |
//|   - Reads from trading EA via extern PositionState[] array       |
//|   - Queries MT5 HistorySelect() for closed trades               |
//|   - Detects bar close via OnCalculate() time[] array            |
//|   - Updates ChartObjects once per bar (D-06)                    |
//|                                                                  |
//| Deployment:                                                      |
//|   Attach to dedicated MT5 chart (not the trading EA chart).      |
//|   EA and indicator run independently.                            |
//|                                                                  |
//+------------------------------------------------------------------+

#property indicator_separate_window
#property indicator_buffers 0
#property indicator_plots   0

#include "Include/Dashboard.mqh"
#include "Include/TradeExecution.mqh"

//+------------------------------------------------------------------+
//| Global Variables                                                 |
//+------------------------------------------------------------------+

DashboardMetrics g_metrics;                    // Master metrics struct
static datetime g_lastBarTime = 0;             // Bar-close detection
static bool g_initialized = false;             // Initialization flag

extern int DashboardXOffset = 50;              // Panel X position (pixels from left)
extern int DashboardYOffset = 100;             // Panel Y position (pixels from top)

//+------------------------------------------------------------------+
//| Custom indicator initialization function (OnInit)                |
//+------------------------------------------------------------------+

int OnInit()
{
    // Initialize indicator properties
    IndicatorSetString(INDICATOR_SHORTNAME, "Volume Profile Dashboard");
    
    // Initialize dashboard metrics
    InitializeDashboard(g_metrics);
    
    // Set initial bar time (will update on first OnCalculate)
    g_lastBarTime = 0;
    g_initialized = true;
    
    Print("Dashboard Indicator initialized. Attach to dedicated chart window.");
    Print(StringFormat("Dashboard position: X=%d, Y=%d", DashboardXOffset, DashboardYOffset));
    
    return(INIT_SUCCEEDED);
}
```

Implement OnCalculate() and UpdateDashboard() functions. These are critical for ensuring bar-close-only updates (D-06 constraint):

```mql5
//+------------------------------------------------------------------+
//| Custom indicator iteration function (OnCalculate)                |
//|                                                                  |
//| D-06: Bar-close refresh cadence                                  |
//| Called once per tick; must detect bar close and update only once |
//+------------------------------------------------------------------+

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const double &high[],
                const double &low[],
                const double &close[],
                const long &tick_volume[],
                const long &volume[],
                const int &spread[])
{
    // CRITICAL: Bar-close detection (D-06 pattern from RESEARCH.md Pattern 1)
    // Only update dashboard when new bar completes, not every tick
    
    if (time[0] != g_lastBarTime)
    {
        // New bar detected; update dashboard
        g_lastBarTime = time[0];
        UpdateDashboard();
    }
    
    return(rates_total);  // Return total number of bars processed
}

//+------------------------------------------------------------------+
//| Update Dashboard (called once per completed bar)                 |
//|                                                                  |
//| Sequence:                                                        |
//|   1. Refresh all metrics via Dashboard.mqh functions             |
//|   2. Format metrics for display                                  |
//|   3. Prepare data for ChartObject rendering (Plan 03)            |
//+------------------------------------------------------------------+

void UpdateDashboard()
{
    // Step 1: Refresh all dashboard metrics (single function call)
    RefreshDashboardMetrics(g_metrics);
    
    // Step 2: Log current state to MT5 Journal (optional, for debugging)
    LogDashboardState();
    
    // Step 3: Update ChartObjects (handled in Plan 03)
    // For now, just signal that data is ready
    // ChartObjects will be created/updated in RenderDashboard() (Plan 03)
}

//+------------------------------------------------------------------+
//| Log Dashboard State (debug/audit trail)                          |
//+------------------------------------------------------------------+

void LogDashboardState()
{
    string msg = StringFormat(
        "Dashboard Update | Time=%s | Equity=%.2f | "
        "Trades=%d (Win=%.1f%%) | PF=%.2f | MaxDD=%.2f%% | "
        "XAUUSD=%d | EURUSD=%d",
        TimeToString(TimeCurrent()),
        g_metrics.currentEquity,
        g_metrics.totalTrades,
        g_metrics.winRate * 100,
        g_metrics.profitFactor,
        g_metrics.maxDailyDrawdown * 100,
        g_metrics.symbolXAUUSD_trades,
        g_metrics.symbolEURUSD_trades
    );
    
    Print(msg);  // Output to MT5 Journal
}
```

**Design Rationale:**

1. **Bar-close detection pattern (D-06):**
   ```mql5
   if (time[0] != g_lastBarTime)
   {
       g_lastBarTime = time[0];
       UpdateDashboard();  // Called once per bar
   }
   ```
   - time[0] is the timestamp of the CURRENT (forming) bar
   - When time[0] != g_lastBarTime, a new bar has formed (previous bar is closed)
   - This pattern ensures UpdateDashboard() runs exactly once per completed bar, not every tick
   - Reference: RESEARCH.md Pitfall 1 — "HistorySelect() Called Every Tick"

2. **UpdateDashboard() orchestration:**
   - Calls RefreshDashboardMetrics(g_metrics) — single master function from Dashboard.mqh
   - This triggers all sub-functions: UpdateEquityCurve, UpdateDailyPnL, CalculateSummaryStats, CalculatePerSymbolStats, CalculateMaxDrawdown
   - g_metrics struct is now populated with current data
   - Plan 03 will read g_metrics and render ChartObjects

3. **LogDashboardState() for audit trail:**
   - Optional but useful for traders to see dashboard updates in MT5 Journal
   - Shows equity, trade count, win rate, profit factor, max DD, per-symbol trade counts
   - One log line per bar close (not excessive)

4. **OnCalculate return value:**
   - Returns rates_total (standard pattern for indicators)
   - Tells MT5 that all bars have been processed

**Critical Pitfall Avoidance (from RESEARCH.md):**

- **Pitfall 1: HistorySelect() Called Every Tick** — Mitigated by bar-close detection. UpdateDashboard() (which calls HistorySelect indirectly via RefreshDashboardMetrics) only fires once per bar.
- **Pitfall 5: Panel Not Updating** — Handled in Plan 03 (rendering functions).

**Validation:** OnCalculate() compiles; bar-close detection logic is correct.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Dashboard Indicator ↔ MT5 Terminal | Reads chart time[] array (trusted) and account data via AccountInfoDouble() (trusted) |
| Dashboard Indicator ↔ Trading EA | May read extern PositionState[] if exposed (trusted if same process) |
| Dashboard Indicator ↔ HistorySelect() | Queries closed trade history (broker-provided, may be delayed; mitigated by once-per-bar updates) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T5-05 | Timing/DoS | HistorySelect() called every tick | mitigate | Bar-close detection ensures HistorySelect() called max once per bar. If bar frequency is high (tick timeframe), call is still 60-100 times per second max, acceptable. |
| T5-06 | Availability | Indicator detached by user error | accept | Indicator is separate window; user can detach/reattach without affecting EA. No critical dependency. |
| T5-07 | Integrity | g_metrics struct corrupted mid-update | accept | g_metrics updated atomically once per bar. No concurrent access. Single-threaded MQL5 process. |

</threat_model>
