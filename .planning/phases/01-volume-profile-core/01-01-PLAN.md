1. **Find price range from 150-bar lookback:**
   ```mql5
   int lookbackPeriod = 150;
   double minPrice = iLowest(Symbol(), PERIOD_CURRENT, MODE_LOW, lookbackPeriod, 0);
   double maxPrice = iHighest(Symbol(), PERIOD_CURRENT, MODE_HIGH, lookbackPeriod, 0);

   ```

2. **Initialize volume array to zero:**
   ```mql5
   ArrayInitialize(currentProfile.volumeArray, 0);
   currentProfile.minPrice = minPrice;
   currentProfile.maxPrice = maxPrice;
   currentProfile.binSize = binSize;
   ```

3. **Iterate through 150 bars, prorate each candle's volume proportionally across bins:**
   - For each bar i from 0 to 149:
     - Get High, Low, Close, Volume
     - Calculate candle range = High - Low
     - **If range > 0 (multi-level candle):**
       - Calculate number of bins touched: `numBins = ceil(range / binSize)`
       - Divide volume evenly: `volumePerBin = candle_volume / numBins`
       - For each price level from Low to High (step by binSize):
         - Find bin index: `binIdx = (price - minPrice) / binSize`
         - Add proportional volume: `volumeArray[binIdx] += volumePerBin`
     - **Else (doji or flat candle):**
       - All volume goes to single bin at Close price
       - `binIdx = (Close - minPrice) / binSize`
       - `volumeArray[binIdx] += candle_volume`

4. **Store metadata in currentProfile struct:**
   ```mql5
   currentProfile.pocPrice = 0;  // Will be calculated in Task 3
   currentProfile.vahPrice = 0;
   currentProfile.valPrice = 0;
   currentProfile.pocBinIndex = 0;
   ```

**Complete MQL5 implementation for CalculateCurrentVolumeProfile():**
```mql5
void CalculateCurrentVolumeProfile()
{
    int lookbackPeriod = 150;
    double minPrice = iLowest(Symbol(), PERIOD_CURRENT, MODE_LOW, lookbackPeriod, 0);
    double maxPrice = iHighest(Symbol(), PERIOD_CURRENT, MODE_HIGH, lookbackPeriod, 0);
    
    if (maxPrice <= minPrice)
    {
        Print("ERROR: Invalid price range for volume profile");
        return;
    }
    
    double binSize = (maxPrice - minPrice) / VOLUME_BINS;
    
    // Store in struct
    currentProfile.minPrice = minPrice;
    currentProfile.maxPrice = maxPrice;
    currentProfile.binSize = binSize;
    
    // Initialize array
    ArrayInitialize(currentProfile.volumeArray, 0);
    
    // Prorate volume from 150 bars
    for (int i = 0; i < lookbackPeriod; i++)
    {
        double high = iHigh(Symbol(), PERIOD_CURRENT, i);
        double low = iLow(Symbol(), PERIOD_CURRENT, i);
        double close = iClose(Symbol(), PERIOD_CURRENT, i);
        long volume = iVolume(Symbol(), PERIOD_CURRENT, i);
        
        if (volume <= 0) continue;  // Skip bars with zero volume
        
        double range = high - low;
        
        if (range > binSize)  // Multi-level candle
        {
            int numBins = (int)(range / binSize) + 1;
            if (numBins > VOLUME_BINS) numBins = VOLUME_BINS;  // Safety cap
            
            double volumePerBin = (double)volume / numBins;
            
            // Iterate from low to high in bin steps
            for (double price = low; price <= high && price <= maxPrice; price += binSize)
            {
                int binIdx = (int)((price - minPrice) / binSize);
                if (binIdx >= 0 && binIdx < VOLUME_BINS)
                {
                    currentProfile.volumeArray[binIdx] += volumePerBin;
                }
            }
        }
        else  // Doji or small range
        {
            int binIdx = (int)((close - minPrice) / binSize);
            if (binIdx >= 0 && binIdx < VOLUME_BINS)
            {
                currentProfile.volumeArray[binIdx] += volume;
            }
        }
    }
    
    // Validation: sum should equal total volume ±0.1%
    double binSum = 0;
    long rawTotal = 0;
    for (int i = 0; i < lookbackPeriod; i++)
        rawTotal += iVolume(Symbol(), PERIOD_CURRENT, i);
    
    for (int i = 0; i < VOLUME_BINS; i++)
        binSum += currentProfile.volumeArray[i];
    
    if (rawTotal > 0)
    {
        double variance = MathAbs(binSum - rawTotal) / rawTotal;
        if (variance > 0.01)  // >1% variance
        {
            Print("WARNING: Volume distribution variance ", variance * 100, "% > 1%");
        }
    }
}
```

**Key points per D-01 (proportional-to-range):**
- Volume is NOT split as fixed 60/40 body/wick
- Volume IS distributed proportionally to the price distance each part of the candle occupies
- This produces more accurate POC/VAH/VAL than fixed splits

**Acceptance per REQ-001 and REQ-008:**
- sum(volumeArray[]) ≈ total tick volume ±0.1%
- Multi-level candles prorate correctly
- Function callable and produces non-zero distribution

**Step 2: VAH/VAL Calculation (70% Value Area Expansion)**
VAH/VAL = boundaries that capture 70% cumulative volume expanding outward from POC (REQ-003, REQ-004).

```mql5
void CalculateValueArea()
{
    if (currentProfile.pocBinIndex < 0)
    {
        Print("ERROR: POC not calculated before VAH/VAL");
        return;
    }
    
    // Step 1: Calculate total volume and target (70%)
    double totalVol = 0;
    for (int i = 0; i < VOLUME_BINS; i++)
    {
        totalVol += currentProfile.volumeArray[i];
    }
    
    if (totalVol <= 0)
    {
        Print("ERROR: Total volume <= 0 for VAH/VAL");
        return;
    }
    
    double targetVol = totalVol * 0.70;  // 70% threshold
    
    // Step 2: Expand outward from POC
    double cumulativeVol = currentProfile.volumeArray[currentProfile.pocBinIndex];
    int offset = 0;
    int maxOffset = 200;  // Safety: don't expand > 50% of bins
    
    while (cumulativeVol < targetVol && offset < maxOffset)
    {
        offset++;
        
        // Add bin above POC (higher price)
        if (currentProfile.pocBinIndex + offset < VOLUME_BINS)
        {
            cumulativeVol += currentProfile.volumeArray[currentProfile.pocBinIndex + offset];
        }
        
        // Add bin below POC (lower price)
        if (currentProfile.pocBinIndex - offset >= 0)
        {
            cumulativeVol += currentProfile.volumeArray[currentProfile.pocBinIndex - offset];
        }
    }
    
    // Step 3: Calculate VAH and VAL prices
    int vahBinIndex = currentProfile.pocBinIndex + offset;
    int valBinIndex = currentProfile.pocBinIndex - offset;
    
    // Clamp to valid range
    if (vahBinIndex >= VOLUME_BINS) vahBinIndex = VOLUME_BINS - 1;
    if (valBinIndex < 0) valBinIndex = 0;
    
    currentProfile.vahPrice = currentProfile.minPrice + (vahBinIndex * currentProfile.binSize);
    currentProfile.valPrice = currentProfile.minPrice + (valBinIndex * currentProfile.binSize);
    
    // Validation: VA width should be reasonable
    double vaWidth = currentProfile.vahPrice - currentProfile.valPrice;
    if (vaWidth < currentProfile.binSize)
    {
        Print("WARNING: VA width = ", vaWidth, " < bin size = ", currentProfile.binSize);
    }
}
```

**Algorithm (from RESEARCH.md):**

```mql5
void IdentifyVolumeNodes()
{
    if (currentProfile.pocPrice <= 0)
    {
        Print("ERROR: Volume profile not calculated before node identification");
        return;
    }
    
    // Step 1: Calculate average volume per bin
    double totalVol = 0;
    for (int i = 0; i < VOLUME_BINS; i++)
    {
        totalVol += currentProfile.volumeArray[i];
    }
    
    double avgVolume = totalVol / VOLUME_BINS;
    
    // Step 2: Calculate thresholds (locked, non-negotiable)
    double hvnThreshold = avgVolume * HVN_MULTIPLIER;  // 1.3x per D-02
    double lvnThreshold = avgVolume * LVN_MULTIPLIER;  // 0.7x per D-02
    
    // Step 3: Reset arrays
    currentProfile.hvnCount = 0;
    currentProfile.lvnCount = 0;
    ArrayInitialize(currentProfile.hvnArray, 0);
    ArrayInitialize(currentProfile.lvnArray, 0);
    
    // Step 4: Classify bins as HVN or LVN
    for (int i = 0; i < VOLUME_BINS; i++)
    {
        double binVolume = currentProfile.volumeArray[i];
        double binPrice = currentProfile.minPrice + (i * currentProfile.binSize);
        
        // HVN: local peaks > 1.3x average
        if (binVolume > hvnThreshold)
        {
            if (currentProfile.hvnCount < 50)  // Max 50 HVN clusters
            {
                currentProfile.hvnArray[currentProfile.hvnCount].price = binPrice;
                currentProfile.hvnArray[currentProfile.hvnCount].volume = binVolume;
                currentProfile.hvnCount++;
            }
        }
        
        // LVN: local valleys < 0.7x average
        if (binVolume < lvnThreshold)
        {
            if (currentProfile.lvnCount < 50)  // Max 50 LVN clusters
            {
                currentProfile.lvnArray[currentProfile.lvnCount].price = binPrice;
                currentProfile.lvnCount++;
            }
        }
    }
    
    // Validation: sanity-check cluster counts
    if (currentProfile.hvnCount > 50)
    {
        Print("WARNING: HVN count exceeds max (50); truncated at ", 
              currentProfile.hvnCount, " clusters");
        currentProfile.hvnCount = 50;
    }
    
    if (currentProfile.lvnCount > 50)
    {
        Print("WARNING: LVN count exceeds max (50); truncated at ", 
              currentProfile.lvnCount, " clusters");
        currentProfile.lvnCount = 50;
    }
}
```

- REQ-006: LVN detected as valleys < 25th percentile (0.7x avg volume)

    for (int i = 0; i < VOLUME_BINS; i++)
        binSum += currentProfile.volumeArray[i];
    
    // Get raw total from 150 bars
    long rawTotal = 0;
    for (int i = 0; i < 150; i++)
        rawTotal += iVolume(Symbol(), PERIOD_CURRENT, i);
    
    if (rawTotal <= 0)
    {
        Print("  SKIP: No volume data available (live/incomplete bar)");
        return true;  // Can't validate without data
    }
    
    double variance = MathAbs(binSum - rawTotal) / rawTotal;
    
    if (variance <= 0.01)  // ±1% tolerance
    {
        Print("  PASS: Volume distribution variance = ", variance * 100, "%");
        return true;
    }
    else
    {
        Print("  FAIL: Volume distribution variance = ", variance * 100, "% > 1%");
        return false;
    }
}
