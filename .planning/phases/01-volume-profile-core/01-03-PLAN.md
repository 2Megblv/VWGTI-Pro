**1. Data Quality Validation Function:**

```mql5
bool ValidateDataQuality()
{
    // Check if price data is available and reasonable
    double high = iHigh(Symbol(), PERIOD_CURRENT, 0);
    double low = iLow(Symbol(), PERIOD_CURRENT, 0);
    double close = iClose(Symbol(), PERIOD_CURRENT, 0);
    long volume = iVolume(Symbol(), PERIOD_CURRENT, 0);
    
    if (high <= 0 || low <= 0 || close <= 0)
    {
        Print("ERROR: Invalid OHLC data - High: ", high, " Low: ", low, " Close: ", close);
        return false;
    }
    
    if (high < low)
    {
        Print("ERROR: High < Low in single bar (data corruption)");
        return false;
    }
    
    if (volume <= 0)
    {
        Print("WARNING: Zero volume on current bar");
        // Don't fail; some brokers have zero-volume bars
    }
    
    if (close < low || close > high)
    {
        Print("WARNING: Close outside High-Low range");
    }
    
    return true;
}
```
